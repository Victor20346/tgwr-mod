country_event = {
	id = ww1_lithuania.1
	
	trigger = {
		tag = LIT
		has_country_flag = released_by_germany
		OR = {
		    is_subject = no
			AND = {
				is_subject_of = GER
				GER = { has_country_flag = great_war_loser }
				NOT = { has_global_flag = versallies_treaty}
			}
        }
	}	
	mean_time_to_happen = {
		days = 5
	}
	
	fire_only_once = yes
	hidden = yes

	immediate = {
		drop_cosmetic_tag = yes
		if = {
			limit = {
				has_country_leader = {
				    name = "Mindaugas II"
				    ruling_only = no
				}
			}
			retire_ideology_leader = neutrality
			set_party_name = {
			    ideology = neutrality 
			    long_name = LIT_neutrality_party_long 
			    name = LIT_neutrality_party
			}		
		}
        if = {
		    limit = {
                is_subject_of = GER
		    }
            GER = { end_puppet = LIT}
            leave_faction = yes			
		}		
	}
}
