# Surrender to Allies
country_event = {
	id = ww1_persia.5 
	title = ww1_persia.5.t
	desc = ww1_persia.5.d
	picture = GFX_report_event_american_hunger_march

	fire_only_once = yes
	trigger = {
		original_tag = PER
		is_subject = no
		has_civil_war = no
		has_war_with = TUR
		any_state = { 
			has_dynamic_modifier = { modifier = per_paramilitary_gendarmerie }
		}
	}
	mean_time_to_happen = { days = 14 }
	option = {
		name = ww1_persia.5.a	#surrender to Britain
		ENG = {
			if = {
				limit = {
					OR = {
						has_dlc = "Together for Victory"
						has_dlc = "Man the Guns"
					}
				}
				
				set_autonomy = {
					target = PER
					autonomous_state = autonomy_tgwr_protectorate
					freedom_level = 0.90
				}
				else = {
					puppet = PER
				}
			}
		}
		hidden_effect = {
			set_popularities = {
				democratic = 18
				neutrality = 75
				fascism = 0
				communism = 7
			}
			set_politics = { ruling_party = neutrality }
			every_state = {
				limit = { 
					has_state_flag = PER_russian_british_cordon 
					NOT = { is_claimed_by = RUS }
					NOT = { is_claimed_by = ENG }
				}
				PER = { transfer_state = PREV }
			}
		}	
		start_civil_war = { 
			ideology = fascism 
			size = 0.5
			states = all			
			states_filter = {
				has_dynamic_modifier = { modifier = per_paramilitary_gendarmerie }
			}
		}
		hidden_effect = {
			random_country = {
				limit = {
					original_tag = PER
					has_government = fascism
				}
				every_other_country = {
					add_ai_strategy = {
						type = protect
						id = PREV
						value = -200
					}
					add_ai_strategy = {
						type = alliance
						id = PREV
						value = -200
					}
				}
				set_popularities = {
					democratic = 35
					neutrality = 10
					fascism = 55
					communism = 0
				}
				set_cosmetic_tag = PER_gendarmerie
			}
		}
		set_global_flag = persian_gendarmerie_revolt
		ai_chance = { factor = 100 }
	}
	option = {
		name = ww1_persia.5.b	#stand with gendarmie
		
		hidden_effect = {
			set_popularities = {
				democratic = 35
				neutrality = 10
				fascism = 55
				communism = 0
			}
			set_politics = { ruling_party = fascism }
			every_state = {
				limit = { 
					has_state_flag = PER_russian_british_cordon 
					NOT = { is_claimed_by = RUS }
					NOT = { is_claimed_by = ENG }
				}
				PER = { transfer_state = PREV }
			}
			every_other_country = {
				add_ai_strategy = {
					type = protect
					id = PER
					value = -200
				}
				add_ai_strategy = {
					type = alliance
					id = PER
					value = -200
				}
			}
		}
		start_civil_war = {
			ideology = neutrality
			size = 0.5
			states = all
			states_filter = {
				has_state_flag = PER_russian_british_cordon
			}
			ENG = {
				if = {
					limit = {
						OR = {
							has_dlc = "Together for Victory"
							has_dlc = "Man the Guns"
						}
					}
					
					set_autonomy = {
						target = PER
						autonomous_state = autonomy_tgwr_protectorate
						freedom_level = 0.90
					}
					else = {
						puppet = PER
					}
				}
			}
		}
		hidden_effect = {
			random_country = {
				limit = {
					original_tag = PER
					has_government = neutrality
				}		
				set_popularities = {
					democratic = 18
					neutrality = 75
					fascism = 0
					communism = 7
				}
			}
		}
		PER = { set_cosmetic_tag = PER_gendarmerie }
		set_global_flag = persian_gendarmerie_revolt
		ai_chance = { factor = 0 }
	}
}
